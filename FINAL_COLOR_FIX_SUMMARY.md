```css
bg-gray-50      → bg-white/5
bg-gray-100     → bg-white/8
bg-gray-200     → bg-white/10

/* الخلفيات الملونة */
bg-red-50       → bg-red-500/20
bg-red-100      → bg-red-500/25
bg-red-200      → bg-red-500/30

bg-green-50     → bg-emerald-500/20
bg-blue-50      → bg-blue-500/20
bg-yellow-50    → bg-amber-500/20
bg-purple-50    → bg-purple-500/20
bg-orange-50    → bg-orange-500/20
bg-indigo-50    → bg-indigo-500/20

/* التدرجات */
bg-gradient-to-r from-blue-50 to-blue-100
→ bg-gradient-to-r from-blue-500/20 to-blue-600/20 backdrop-filter backdrop-blur-sm
```
